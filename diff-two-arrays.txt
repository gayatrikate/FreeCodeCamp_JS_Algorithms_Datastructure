function diffArray(arr1, arr2) {
  var newArr = [];
  function diff(a, b){
    for(let i=0; i<a.length; i++){
      if(b.indexOf(a[i])===-1){
        newArr.push(a[i]);
      }
    }
  }
  diff(arr1,arr2);
  diff(arr2,arr1);

  console.log(newArr);
  return newArr;
}

diffArray([1, 2, 3, 5], [1, 2, 3, 4, 5]);
